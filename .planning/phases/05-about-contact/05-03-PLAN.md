```tsx
function buildMapEmbedUrl(address: typeof siteConfig.address): string {
  const hasRealStreet = !address.street.startsWith('//') && !address.street.toLowerCase().includes('placeholder');
  const q = hasRealStreet
    ? `${address.street}, ${address.city}, ${address.state} ${address.zip}`
    : `${address.city}, ${address.state} ${address.zip}`;
  return `https://maps.google.com/maps?q=${encodeURIComponent(q)}&output=embed`;
}
```
Render:
```tsx
<iframe

  title="Raptor Roofing location map"
  loading="lazy"

  style={{ border: 0 }}
  allowFullScreen
  referrerPolicy="no-referrer-when-downgrade"
  className="..."
/>
```
The `title` attribute is mandatory for accessibility. `loading="lazy"` defers the iframe until viewport entry (helps LCP).
