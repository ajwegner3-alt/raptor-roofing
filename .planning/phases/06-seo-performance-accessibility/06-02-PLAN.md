Script contents:

```javascript
#!/usr/bin/env node
// Regression guard: no Review or AggregateRating JSON-LD may ship while
// placeholder testimonials exist. See .planning/phases/06-seo-performance-accessibility/06-CONTEXT.md
// "Schema + FTC/placeholder handling".
import { execSync } from "node:child_process";

const forbidden = [
  { label: "Review JSON-LD",          pattern: '"@type":\\s*"Review"' },
  { label: "AggregateRating JSON-LD", pattern: '"@type":\\s*"AggregateRating"' },
  { label: "reviewBody field",        pattern: "reviewBody" },
  { label: "itemReviewed field",      pattern: "itemReviewed" },
];

let failed = false;

for (const { label, pattern } of forbidden) {
  try {
    // grep -rEn returns 0 on match, 1 on no match, 2+ on error.
    // We want 1 (no match) — a 0 exit means found something forbidden.
    const out = execSync(
      `grep -rEn --include=*.ts --include=*.tsx "${pattern}" src/ || true`,
      { encoding: "utf8" }
    );
    // Filter out comment-only lines (// or * leading).
    const lines = out
      .split("\n")
      .map((l) => l.trim())
      .filter(Boolean)
      .filter((l) => {
        // Strip the "path:line:" prefix to examine the code content
        const code = l.replace(/^[^:]+:\d+:/, "").trim();
        return !code.startsWith("//") && !code.startsWith("*") && !code.startsWith("/*");
      });

    if (lines.length > 0) {
      console.error(`\n❌ FORBIDDEN SCHEMA FOUND — ${label}:`);
      for (const line of lines) console.error(`   ${line}`);
      failed = true;
    }
  } catch (e) {
    console.error(`[check-no-review-schema] grep error for ${label}:`, e.message);
    failed = true;
  }
}

if (failed) {
  console.error(
    "\nThese types must NOT be emitted while placeholder testimonials exist."
  );
  console.error(
    "See .planning/phases/06-seo-performance-accessibility/06-CONTEXT.md > Schema + FTC handling."
  );
  process.exit(1);
}

console.log("✓ check-no-review-schema: OK (no Review/AggregateRating JSON-LD emitted)");
```

Notes:
- File extension is `.mjs` so it runs as ESM regardless of the project's package.json `type` field.
- Uses Node built-ins only — no devDependencies added.
- `grep ... || true` prevents the `|| true` from exiting early on no-match (grep returns 1 on no match, which bash treats as failure).
- The comment-line filter is crude but correct for this codebase — it skips leading `//`, `*`, `/*` lines so the already-present commented-out aggregateRating in `src/lib/schema.tsx` does NOT trigger a false positive.
- Exit code 1 on failure, 0 on pass.
