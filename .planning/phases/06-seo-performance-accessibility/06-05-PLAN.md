Create `scripts/check-bundle-size.mjs` implementing the bundle-size parser from RESEARCH.md Q13. This script runs `npm run build`, parses the route table, and exits 1 if any route's First Load JS exceeds 200 KB.

**Script contents:**

```javascript
#!/usr/bin/env node
// Bundle size gate — enforces 200 KB First Load JS ceiling per route.
// See .planning/phases/06-seo-performance-accessibility/06-CONTEXT.md > Performance budget.
import { execSync } from "node:child_process";

const MAX_KB = 200;

console.log(`[check-bundle-size] Running npm run build (this takes ~30s)...\n`);

let output;
try {
  output = execSync("npm run build 2>&1", { encoding: "utf8", maxBuffer: 16 * 1024 * 1024 });
} catch (err) {
  console.error("[check-bundle-size] npm run build FAILED. Output:\n");
  console.error(err.stdout || err.message);
  process.exit(2);
}

// Next.js 16 build output format (example line):
//   ┌ ○ /                                        8.23 kB         123 kB
//   ├ ○ /about                                   4.12 kB         108 kB
//   ├ λ /api/contact                             0 B              88.2 kB
//   └ ● /services/[slug]                         6.1 kB           115 kB
// We parse route symbol + path + First Load JS (the last number with "kB").
const routeRegex = /[┌├└│]?\s*[○●λƒ]\s+(\S+)\s+[\d.]+\s*k?B\s+([\d.]+)\s*kB/g;

const routes = [];
let match;
while ((match = routeRegex.exec(output)) !== null) {
  const [, route, firstLoadStr] = match;
  routes.push({ route, firstLoadKb: parseFloat(firstLoadStr) });
}

if (routes.length === 0) {
  console.error("[check-bundle-size] Could not parse any routes from build output.");
  console.error("This usually means the Next.js output format changed. Raw output:\n");
  console.error(output.slice(output.indexOf("Route") ?? 0, -1));
  process.exit(3);
}

console.log(`[check-bundle-size] Parsed ${routes.length} routes. Bundle ceiling: ${MAX_KB} kB First Load JS.\n`);

const failures = [];
const passes = [];
for (const { route, firstLoadKb } of routes) {
  const status = firstLoadKb > MAX_KB ? "FAIL" : "OK";
  const line = `  ${status.padEnd(4)} ${route.padEnd(40)} ${firstLoadKb.toFixed(1)} kB`;
  if (firstLoadKb > MAX_KB) {
    failures.push(line);
  } else {
    passes.push(line);
  }
}

console.log("Route table:");
for (const line of [...passes, ...failures]) console.log(line);

if (failures.length > 0) {
  console.error(`\n[check-bundle-size] GATE FAILED — ${failures.length} route(s) exceed ${MAX_KB} kB First Load JS:`);
  for (const line of failures) console.error(line);
  console.error("\nOptions:");
  console.error("  1. Reduce client-side JS on the offending route (audit for unnecessary 'use client')");
  console.error("  2. Lazy-load third-party scripts (e.g., reCAPTCHA on /contact)");
  console.error("  3. Split heavy dependencies via dynamic import()");
  process.exit(1);
}

console.log(`\n[check-bundle-size] PASS — all routes under ${MAX_KB} kB.`);
```

Notes:
- Uses Node built-ins only — no devDeps.
- `maxBuffer: 16 * 1024 * 1024` handles the large-ish Next.js build stdout.
- The regex is lenient about the route-symbol characters `○ ● λ ƒ` (Next.js uses these for Static / SSG-with-data / Dynamic / Function). Adjust if the build table uses different symbols in v16.2.
- Exit codes: 0 = pass, 1 = bundle gate failed, 2 = build failed, 3 = couldn't parse output (format change).

**Step 3 — Run Lighthouse per route.**
Use `npx lighthouse` (no install — uses cached ~/.npm copy). Shell loop:

```bash
OUT_DIR=".planning/phases/06-seo-performance-accessibility/lighthouse-reports"
BASE="http://localhost:3000"

declare -A ROUTES
ROUTES["home"]="/"
ROUTES["about"]="/about"
ROUTES["contact"]="/contact"
ROUTES["services-roofing"]="/services/roofing"
ROUTES["services-siding"]="/services/siding"
ROUTES["services-gutters"]="/services/gutters"
ROUTES["services-emergency-tarping"]="/services/emergency-tarping"

for name in "${!ROUTES[@]}"; do
  path="${ROUTES[$name]}"
  url="${BASE}${path}"
  out="${OUT_DIR}/lighthouse-${name}.html"
  echo ">>> Auditing $url"
  npx lighthouse "$url" \
    --preset=desktop \
    --form-factor=mobile \
    --throttling-method=simulate \
    --output=html \
    --output-path="$out" \
    --only-categories=performance,accessibility,best-practices,seo \
    --chrome-flags="--headless=new --no-sandbox" \
    --quiet
done
```
